class Mythread extends Thread
    {
    public void run()
    {
        System.out.println(getName()+" Hello World!");
    }
    }
class Thread1 {
    public static void main(String[] args) {
        Mythread t1 = new Mythread();
        Mythread t2 = new Mythread();
        Mythread t3 = new Mythread();
        Mythread t4 = new Mythread();
        t1.setName("T1");
        t2.setName("T2");
        t3.setName("T3");
        t4.setName("T4");
        t2.start();
        t1.start();
        t3.start();
        t4.start();
    }
            
}

Output:

T2 Hello World!
T3 Hello World!
T1 Hello World!
T4 Hello World!
