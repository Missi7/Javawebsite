{
    public void run()
    {
        for(int i=1;i<=20;i++)
        {
            System.out.println(i);
        }
    }
}
class Thread2 {
public static void main(String[] args) {
    Mythread t1 = new Mythread();
    Mythread t2 = new Mythread();
    t1.start();
    t2.start();
}
}
