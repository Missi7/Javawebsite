public class StringToArray {
    public static void main(String[] args) {
        String str = "Hello World";
        char[] charArray = new char[str.length()];

        for (int i = 0; i < str.length(); i++) {
            charArray[i] = str.charAt(i);
        }

        for (char c : charArray) {
            System.out.print(c + " ");
        }
    }
}

Output:

H e l l o   W o r l d
