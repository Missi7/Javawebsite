class Main{
    static String strReverse(String str){
        String revstr = "";
        
        for(int i = (str.length()-1); i >= 0; i--){
            revstr += str.charAt(i);
        }
        return revstr;
    }
    public static void main(String[] args){
        System.out.println(strReverse("Hello"));
    }
}

Output:

olleH
