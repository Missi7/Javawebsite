class Main{
    static String replaceSpaces(String str, char ch){
        String newstr = "";
        for(int i = 0; i < str.length(); i++){
            if(str.charAt(i) == ' '){
                newstr += ch;
            }
            else{
                newstr += str.charAt(i);
            }
        }
        return newstr;
    }
    public static void main(String[] args){
        System.out.println(replaceSpaces("Hello World I am Misbah  ", '*'));
    }
}

Output:

Hello*World*I*am*Misbah**
