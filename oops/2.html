class Dog{
private String name,breed;
Dog(String dName,String dBreed){
    name=dName;
    breed=dBreed;
}
String setName(String dName)
{
    name=dName;
    return name;
}
String setBreed(String dBreed)
{
    breed=dBreed;
    return breed;
}
void display(){
    System.out.println("Dog Name : "+name+"Dog Breed : "+breed);
}
public static void main(String st[])
{
    Dog obj = new Dog("Tiger","Boxer");
    obj.setName("Sheru");
    obj.setBreed("Bulldog");
    obj.display();
}
}

Output:

Dog Name : Sheru
Dog Breed : Bulldog
