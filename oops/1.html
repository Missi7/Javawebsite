class Person{
private String name;
private int age;
Person(String pName , int pAge){
    name = pName;
    age = pAge;
}
void display(){
    System.out.println("Name : "+name+"Age : "+age);
}
public static void main(String st[])
{
    Person obj1=new Person("Misbah",20);
    obj1.display();
    Person obj2=new Person("Suhail",22);
    obj2.display();
}
}

Output:

Name : Manjit
Age : 18
Name : Sandeep
Age : 24
