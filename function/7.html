class Main{
static Boolean isPasswordValid(String str){
    int len = str.length(), charlen = 0;
    boolean hasnum = false, hassymbol = false;
        
    if(len < 12){
    return false;
}
    for(int i = 0; i <len; i++){
    if(str.charAt(i) >= 'A' && str.charAt(i) <= 'Z' || str.charAt(i) >= 'a' && str.charAt(i) <= 'z'){
    charlen++;
}
    else if(str.charAt(i) >= '0' && str.charAt(i) <= '9'){
        hasnum = true;
    }
    else if(str.charAt(i) >= '!' && str.charAt(i) <= '@'){
        hassymbol = true;	
    }
}
    if(charlen >=8 && hasnum && hassymbol){
        return true;
    }
    return false;
}
public static void main(String[] args){
System.out.println(isPasswordValid("Misbah@42261") ? "Password is Valid" : "Invalid Password" );
System.out.println(isPasswordValid("Misbah4226") ? "Password is Valid" : "Invalid Password" );
}
}
