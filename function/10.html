public class NumDigit {
static void numDigit(int num)
{
    int count=0,temp;
    while (num>0) {
        temp = num%10;
        if (temp==2) {
            count++;
        } 
        num/=10;
}
    System.out.println("count "+count);
}
    public static void main(String[] args) {
    numDigit(12322);
    }
}

Output:

count 3
