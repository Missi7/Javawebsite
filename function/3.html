public class MidChar {
    static void midChar(String s)
    {
    if ((s.length()%2)==0) {
    int mid1 = ((s.length())/2)-1;
    int mid2 = (s.length()/2);
    System.out.println(s.charAt(mid1));
    System.out.println(s.charAt(mid2));
    }
    else
    {
    int mid = ((s.length())/2);
    System.out.println(s.charAt(mid));
    }
}
public static void main(String[] args) {
    String str = "Misbah";
    midChar(str);
    }
}
