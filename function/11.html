public class ThreeInt {
boolean consecutive(int a,int b,int c)
{
    if (++a==b && ++b==c) {
        return true;
    }
    else
    {
        return false;
    }
                
}
public static void main(String[] args) {
    int num1=2,num2=3,num3=4;
    boolean cons = new ThreeInt().consecutive(num1, num2, num3);
    System.out.println(cons);
    }
}

Output:

true
